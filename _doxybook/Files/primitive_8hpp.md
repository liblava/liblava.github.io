#pragma once

#include "liblava/util/math.hpp"

namespace lava {

struct vertex {
    using list = std::vector<vertex>;

    v3 position;

    v4 color;

    v2 uv;

    v3 normal;

    bool operator==(vertex const& other) const {
        return position == other.position
               && color == other.color
               && uv == other.uv
               && normal == other.normal;
    }
};

enum class mesh_type : index {
    none = 0,
    cube,
    triangle,
    quad,
    hexagon,
};

} // namespace lava
