357//-----------------------------------------------------------------------------
358template<typename T>
359bool mesh_template<T>::create(device_p d,
360 bool m,
361 VmaMemoryUsage mu) {
362 device = d;
363 mapped = m;
364 memory_usage = mu;
365
366 if (!data.vertices.empty()) {
367 vertex_buffer = buffer::make();
368
369 if (!vertex_buffer->create(device,
370 data.vertices.data(),
371 sizeof(T) * data.vertices.size(),
372 VK_BUFFER_USAGE_VERTEX_BUFFER_BIT,
373 mapped,
374 memory_usage)) {
375 log()->error("create mesh vertex buffer");
376 return false;
377 }
378 }
379
380 if (!data.indices.empty()) {
381 index_buffer = buffer::make();
382
383 if (!index_buffer->create(device,
384 data.indices.data(),
385 sizeof(ui32) * data.indices.size(),
386 VK_BUFFER_USAGE_INDEX_BUFFER_BIT,
387 mapped,
388 memory_usage)) {
389 log()->error("create mesh index buffer");
390 return false;
391 }
392 }
393
394 return true;
395}

460//-----------------------------------------------------------------------------
461template<typename T,
462 bool generate_colors,
463 bool generate_normals,
464 bool generate_uvs,
465 bool has_colors,
466 bool has_normals,
467 bool has_uvs>
468mesh_template_data<T> create_mesh_data(mesh_type type) {
469 mesh_template_data<T> return_mesh_data;
470
471 constexpr bool auto_position = requires(const T t) {
472 t.position;
473 };
474 static_assert(auto_position,
475 "Vertex struct `T` must contain field `position`");
476
477 constexpr bool auto_colors = requires(const T t) {
478 t.color;
479 };
480 constexpr bool auto_normals = requires(const T t) {
481 t.normal;
482 };
483 constexpr bool auto_uvs = requires(const T t) {
484 t.uv;
485 };
486
487 using PosType = decltype(T::position);
488
489 switch (type) {
490 case mesh_type::cube: {
491 return_mesh_data.indices.reserve(36);
492 return_mesh_data.indices = make_primitive_indices_cube<(generate_normals
493 && auto_normals)>();
494 constexpr size_t vert_count = (generate_normals && auto_normals) ? 24 : 8;
495 return_mesh_data.vertices.reserve(vert_count);
496 auto positions = make_primitive_positions_cube<PosType, vert_count,
497 (generate_normals && auto_normals)>();
498 for (size_t i = 0; i < vert_count; i++) {
499 T vert;
500 vert.position = positions[i];
501 if constexpr (generate_normals && auto_normals) {
502 // this array is generated inside of every loop because
503 // that makes the scoping rules simplest to follow
504 // my expectation is that a compiler should be able
505 // to trivially optimize this
506 using NormType = decltype(T::normal);
507 auto normals = make_primitive_normals_cube<NormType>();
508 vert.normal = normals[i / 4];
509 }
510 if constexpr (generate_uvs && auto_uvs) {
511 using UVType = decltype(T::uv);
512 auto uvs = make_primitive_uvs_cube<UVType>();
513 vert.uv = uvs[i];
514 }
515 return_mesh_data.vertices.push_back(vert);
516 }
517 break;
518 }
519
520 case mesh_type::triangle: {
521 return_mesh_data.vertices.reserve(3);
522 T vert_one;
523 vert_one.position = { 1, 1, 0 };
524 T vert_two;
525 vert_two.position = { -1, 1, 0 };
526 T vert_three;
527 vert_three.position = { 0, -1, 0 };
528 if constexpr (generate_uvs && auto_uvs) {
529 vert_one.uv = { 1, 1 };
530 vert_two.uv = { 0, 1 };
531 vert_three.uv = { 0.5, 0 };
532 }
533 if constexpr (generate_normals && auto_normals) {
534 vert_one.normal = { 1, 1, 0 };
535 vert_two.normal = { -1, 1, 0 };
536 vert_three.normal = { 0, -1, 0 };
537 }
538 return_mesh_data.vertices.push_back(vert_one);
539 return_mesh_data.vertices.push_back(vert_two);
540 return_mesh_data.vertices.push_back(vert_three);
541 break;
542 }
543
544 case mesh_type::quad: {
545 return_mesh_data.vertices.reserve(4);
546 return_mesh_data.indices.reserve(6);
547 T vert_one;
548 vert_one.position = { 1, 1, 0 };
549 T vert_two;
550 vert_two.position = { -1, 1, 0 };
551 T vert_three;
552 vert_three.position = { -1, -1, 0 };
553 T vert_four;
554 vert_four.position = { 1, -1, 0 };
555 if constexpr (generate_uvs && auto_uvs) {
556 vert_one.uv = { 1, 1 };
557 vert_two.uv = { 0, 1 };
558 vert_three.uv = { 0, 0 };
559 vert_four.uv = { 1, 0 };
560 }
561 if constexpr (generate_normals && auto_normals) {
562 vert_one.normal = { 0, 0, 1 };
563 vert_two.normal = { 0, 0, 1 };
564 vert_three.normal = { 0, 0, 1 };
565 vert_four.normal = { 0, 0, 1 };
566 }
567 // clang-format off
568 return_mesh_data.indices = {
569 0, 1, 2,
570 2, 3, 0,
571 };
572 // clang-format on
573 return_mesh_data.vertices.push_back(vert_one);
574 return_mesh_data.vertices.push_back(vert_two);
575 return_mesh_data.vertices.push_back(vert_three);
576 return_mesh_data.vertices.push_back(vert_four);
577 break;
578 }
579
580 case mesh_type::hexagon: {
581 return_mesh_data.vertices.reserve(7);
582 return_mesh_data.indices.reserve(18);
583 hex_layout layout;
584 layout.orientation = hex_layout_point_y;
585 layout.size = { 1, 1 };
586 auto hex_corners = hex_polygon_corners(layout, {});
587 T vert_center;
588 vert_center.position = { 0, 0, 0 };
589 T vert_sw;
590 PosType temp;
591 temp[0] = hex_corners.at(0).x;
592 temp[1] = hex_corners.at(0).y;
593 temp[2] = 0;
594 vert_sw.position = temp;
595 T vert_nw;
596 temp[0] = hex_corners.at(1).x;
597 temp[1] = hex_corners.at(1).y;
598 vert_nw.position = temp;
599 T vert_n;
600 temp[0] = hex_corners.at(2).x;
601 temp[1] = hex_corners.at(2).y;
602 vert_n.position = temp;
603 T vert_ne;
604 temp[0] = hex_corners.at(3).x;
605 temp[1] = hex_corners.at(3).y;
606 vert_ne.position = temp;
607 T vert_se;
608 temp[0] = hex_corners.at(4).x;
609 temp[1] = hex_corners.at(4).y;
610 vert_se.position = temp;
611 T vert_s;
612 temp[0] = hex_corners.at(5).x;
613 temp[1] = hex_corners.at(5).y;
614 vert_s.position = temp;
615 if constexpr (generate_uvs && auto_uvs) {
616 vert_center.uv = { 0, 0 };
617 vert_sw.uv = { 1, 0 };
618 vert_nw.uv = { 0, 1 };
619 vert_n.uv = { 1, 1 };
620 vert_ne.uv = { 1, 0 };
621 vert_se.uv = { 0, 1 };
622 vert_s.uv = { 1, 1 };
623 }
624 if constexpr (generate_normals && auto_normals) {
625 vert_center.normal = { 0, 0, 1 };
626 vert_sw.normal = { 0, 0, 1 };
627 vert_nw.normal = { 0, 0, 1 };
628 vert_n.normal = { 0, 0, 1 };
629 vert_ne.normal = { 0, 0, 1 };
630 vert_se.normal = { 0, 0, 1 };
631 vert_s.normal = { 0, 0, 1 };
632 }
633 // clang-format off
634 return_mesh_data.indices = {
635 0, 1, 6, 0, 6, 5, 0, 5, 4, 0, 4, 3, 0, 3, 2, 0, 2, 1
636 };
637 // clang-format on
638 return_mesh_data.vertices.push_back(vert_center);
639 return_mesh_data.vertices.push_back(vert_sw);
640 return_mesh_data.vertices.push_back(vert_nw);
641 return_mesh_data.vertices.push_back(vert_n);
642 return_mesh_data.vertices.push_back(vert_ne);
643 return_mesh_data.vertices.push_back(vert_se);
644 return_mesh_data.vertices.push_back(vert_s);
645 break;
646 }
647
648 case mesh_type::none:
649 default:
650 break;
651 }
652
653 if constexpr (generate_colors && auto_colors) {
654 for (auto& vert : return_mesh_data.vertices) {
655 // this does not work on glm vectors
656 // for (auto& this_color : vert.color) {
657 // for (auto& color_component : this_color) {
658 // color_component = 1;
659 // }
660 // }
661 if constexpr (std::is_same_v<decltype(vert.color), glm::vec3>) {
662 vert.color = { 1, 1, 1 };
663 } else if constexpr (std::is_same_v<decltype(vert.color), glm::vec4>) {
664 vert.color = { 1, 1, 1, 1 };
665 } else {
666 for (size_t i = 0; i < vert.color.size(); i++) {
667 vert.color[i] = 1;
668 }
669 }
670 }
671 }
672
673 return return_mesh_data;
674}
