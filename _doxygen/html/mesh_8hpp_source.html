356//-----------------------------------------------------------------------------
357template <typename T>
358bool mesh_template<T>::create(device::ptr dev,
359 bool m,
360 VmaMemoryUsage mu) {
361 m_device = dev;
362 m_mapped = m;
363 m_memory_usage = mu;
364
365 if (!m_data.vertices.empty()) {
366 m_vertex_buffer = buffer::make();
367
368 if (!m_vertex_buffer->create(m_device,
369 m_data.vertices.data(),
370 sizeof(T) * m_data.vertices.size(),
371 VK_BUFFER_USAGE_VERTEX_BUFFER_BIT,
372 m_mapped,
373 m_memory_usage)) {
374 logger()->error("create mesh vertex buffer");
375 return false;
376 }
377 }
378
379 if (!m_data.indices.empty()) {
380 m_index_buffer = buffer::make();
381
382 if (!m_index_buffer->create(m_device,
383 m_data.indices.data(),
384 sizeof(ui32) * m_data.indices.size(),
385 VK_BUFFER_USAGE_INDEX_BUFFER_BIT,
386 m_mapped,
387 m_memory_usage)) {
388 logger()->error("create mesh index buffer");
389 return false;
390 }
391 }
392
393 return true;
394}

459//-----------------------------------------------------------------------------
460template <typename T,
461 bool generate_colors,
462 bool generate_normals,
463 bool generate_uvs,
464 bool has_colors,
465 bool has_normals,
466 bool has_uvs>
467mesh_template_data<T> create_mesh_data(mesh_type type) {
468 mesh_template_data<T> return_mesh_data;
469
470 constexpr bool auto_position = requires(const T t) {
471 t.position;
472 };
473 static_assert(auto_position,
474 "Vertex struct `T` must contain field `position`");
475
476 constexpr bool auto_colors = requires(const T t) {
477 t.color;
478 };
479 constexpr bool auto_normals = requires(const T t) {
480 t.normal;
481 };
482 constexpr bool auto_uvs = requires(const T t) {
483 t.uv;
484 };
485
486 using PosType = decltype(T::position);
487
488 switch (type) {
489 case mesh_type::cube: {
490 return_mesh_data.indices.reserve(36);
491 return_mesh_data.indices = make_primitive_indices_cube<(generate_normals
492 && auto_normals)>();
493 constexpr size_t vert_count = (generate_normals && auto_normals) ? 24 : 8;
494 return_mesh_data.vertices.reserve(vert_count);
495 auto positions = make_primitive_positions_cube<PosType, vert_count,
496 (generate_normals && auto_normals)>();
497 for (size_t i = 0; i < vert_count; i++) {
498 T vert;
499 vert.position = positions[i];
500 if constexpr (generate_normals && auto_normals) {
501 // this array is generated inside of every loop because
502 // that makes the scoping rules simplest to follow
503 // my expectation is that a compiler should be able
504 // to trivially optimize this
505 using NormType = decltype(T::normal);
506 auto normals = make_primitive_normals_cube<NormType>();
507 vert.normal = normals[i / 4];
508 }
509 if constexpr (generate_uvs && auto_uvs) {
510 using UVType = decltype(T::uv);
511 auto uvs = make_primitive_uvs_cube<UVType>();
512 vert.uv = uvs[i];
513 }
514 return_mesh_data.vertices.push_back(vert);
515 }
516 break;
517 }
518
519 case mesh_type::triangle: {
520 return_mesh_data.vertices.reserve(3);
521 T vert_one;
522 vert_one.position = {1, 1, 0};
523 T vert_two;
524 vert_two.position = {-1, 1, 0};
525 T vert_three;
526 vert_three.position = {0, -1, 0};
527 if constexpr (generate_uvs && auto_uvs) {
528 vert_one.uv = {1, 1};
529 vert_two.uv = {0, 1};
530 vert_three.uv = {0.5, 0};
531 }
532 if constexpr (generate_normals && auto_normals) {
533 vert_one.normal = {1, 1, 0};
534 vert_two.normal = {-1, 1, 0};
535 vert_three.normal = {0, -1, 0};
536 }
537 return_mesh_data.vertices.push_back(vert_one);
538 return_mesh_data.vertices.push_back(vert_two);
539 return_mesh_data.vertices.push_back(vert_three);
540 break;
541 }
542
543 case mesh_type::quad: {
544 return_mesh_data.vertices.reserve(4);
545 return_mesh_data.indices.reserve(6);
546 T vert_one;
547 vert_one.position = {1, 1, 0};
548 T vert_two;
549 vert_two.position = {-1, 1, 0};
550 T vert_three;
551 vert_three.position = {-1, -1, 0};
552 T vert_four;
553 vert_four.position = {1, -1, 0};
554 if constexpr (generate_uvs && auto_uvs) {
555 vert_one.uv = {1, 1};
556 vert_two.uv = {0, 1};
557 vert_three.uv = {0, 0};
558 vert_four.uv = {1, 0};
559 }
560 if constexpr (generate_normals && auto_normals) {
561 vert_one.normal = {0, 0, 1};
562 vert_two.normal = {0, 0, 1};
563 vert_three.normal = {0, 0, 1};
564 vert_four.normal = {0, 0, 1};
565 }
566 // clang-format off
567 return_mesh_data.indices = {
568 0, 1, 2,
569 2, 3, 0,
570 };
571 // clang-format on
572 return_mesh_data.vertices.push_back(vert_one);
573 return_mesh_data.vertices.push_back(vert_two);
574 return_mesh_data.vertices.push_back(vert_three);
575 return_mesh_data.vertices.push_back(vert_four);
576 break;
577 }
578
579 case mesh_type::hexagon: {
580 return_mesh_data.vertices.reserve(7);
581 return_mesh_data.indices.reserve(18);
582 hex_layout layout;
583 layout.orientation = hex_layout_point_y;
584 layout.size = {1, 1};
585 auto hex_corners = hex_polygon_corners(layout, {});
586 T vert_center;
587 vert_center.position = {0, 0, 0};
588 T vert_sw;
589 PosType temp;
590 temp[0] = hex_corners.at(0).x;
591 temp[1] = hex_corners.at(0).y;
592 temp[2] = 0;
593 vert_sw.position = temp;
594 T vert_nw;
595 temp[0] = hex_corners.at(1).x;
596 temp[1] = hex_corners.at(1).y;
597 vert_nw.position = temp;
598 T vert_n;
599 temp[0] = hex_corners.at(2).x;
600 temp[1] = hex_corners.at(2).y;
601 vert_n.position = temp;
602 T vert_ne;
603 temp[0] = hex_corners.at(3).x;
604 temp[1] = hex_corners.at(3).y;
605 vert_ne.position = temp;
606 T vert_se;
607 temp[0] = hex_corners.at(4).x;
608 temp[1] = hex_corners.at(4).y;
609 vert_se.position = temp;
610 T vert_s;
611 temp[0] = hex_corners.at(5).x;
612 temp[1] = hex_corners.at(5).y;
613 vert_s.position = temp;
614 if constexpr (generate_uvs && auto_uvs) {
615 vert_center.uv = {0, 0};
616 vert_sw.uv = {1, 0};
617 vert_nw.uv = {0, 1};
618 vert_n.uv = {1, 1};
619 vert_ne.uv = {1, 0};
620 vert_se.uv = {0, 1};
621 vert_s.uv = {1, 1};
622 }
623 if constexpr (generate_normals && auto_normals) {
624 vert_center.normal = {0, 0, 1};
625 vert_sw.normal = {0, 0, 1};
626 vert_nw.normal = {0, 0, 1};
627 vert_n.normal = {0, 0, 1};
628 vert_ne.normal = {0, 0, 1};
629 vert_se.normal = {0, 0, 1};
630 vert_s.normal = {0, 0, 1};
631 }
632 // clang-format off
633 return_mesh_data.indices = {
634 0, 1, 6, 0, 6, 5, 0, 5, 4, 0, 4, 3, 0, 3, 2, 0, 2, 1
635 };
636 // clang-format on
637 return_mesh_data.vertices.push_back(vert_center);
638 return_mesh_data.vertices.push_back(vert_sw);
639 return_mesh_data.vertices.push_back(vert_nw);
640 return_mesh_data.vertices.push_back(vert_n);
641 return_mesh_data.vertices.push_back(vert_ne);
642 return_mesh_data.vertices.push_back(vert_se);
643 return_mesh_data.vertices.push_back(vert_s);
644 break;
645 }
646
647 case mesh_type::none:
648 default:
649 break;
650 }
651
652 if constexpr (generate_colors && auto_colors) {
653 for (auto& vert : return_mesh_data.vertices) {
654 // this does not work on glm vectors
655 // for (auto& this_color : vert.color) {
656 // for (auto& color_component : this_color) {
657 // color_component = 1;
658 // }
659 // }
660 if constexpr (std::is_same_v<decltype(vert.color), glm::vec3>) {
661 vert.color = {1, 1, 1};
662 } else if constexpr (std::is_same_v<decltype(vert.color), glm::vec4>) {
663 vert.color = {1, 1, 1, 1};
664 } else {
665 for (size_t i = 0; i < vert.color.size(); i++) {
666 vert.color[i] = 1;
667 }
668 }
669 }
670 }
671
672 return return_mesh_data;
673}
