180inline mat4 perspective_matrix(uv2 size,
181 r32 fov = 90.f,
182 r32 far_plane = 5.f) {
183 // Vulkan NDC is right-handed with Y pointing down
184 // we flip Y which makes it left-handed
185 return glm::scale(glm::identity<glm::mat4>(),
186 {1.f, -1.f, 1.f})
187 * glm::perspectiveLH_ZO(
188 glm::radians(fov),
189 r32(size.x) / size.y,
190 0.1f,
191 far_plane);
192}
