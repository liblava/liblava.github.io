83inline string to_string(version const& version) {
84 string stage_str = to_string(version.stage);
85 if ((version.rev > 1) && (version.stage != version_stage::release))
86 stage_str += fmt::format(" {}", version.rev);
87
88 if (version.release == 0) {

92 return fmt::format("{} {}",

94 } else
95 return fmt::format("{}.{} {}",
