175inline string timestamp(const typename CLOCK_TYPE::time_point& time_point,
176 string_ref format = "%Y-%m-%d %H-%M-%S") {
177 auto ms = std::chrono::duration_cast<milliseconds>(
178 time_point.time_since_epoch())
179 % 1000;
180
181 auto const t = CLOCK_TYPE::to_time_t(time_point);
182 auto const tm = *std::localtime(std::addressof(t));
183
184 std::ostringstream stm;
185 stm << std::put_time(std::addressof(tm), str(format))
186 << '.' << std::setfill('0') << std::setw(3) << ms.count();
187 return stm.str();
188}
