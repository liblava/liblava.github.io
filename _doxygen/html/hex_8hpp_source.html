309inline hex_cell hex_round(hex_frac_cell const& cell) {
310 auto qi = to_i32(std::round(cell.q));
311 auto ri = to_i32(std::round(cell.r));
312 auto si = to_i32(std::round(cell.s));
313
314 auto const q_diff = std::abs(qi - cell.q);
315 auto const r_diff = std::abs(ri - cell.r);
316 auto const s_diff = std::abs(si - cell.s);
317
318 if ((q_diff > r_diff) && (q_diff > s_diff))
319 qi = -ri - si;
320 else if (r_diff > s_diff)
321 ri = -qi - si;
322 else
323 si = -qi - ri;
324
325 return {qi, ri, si};
326}

350inline hex_cell::list hex_line(hex_cell const& a,
351 hex_cell const& b) {
352 auto const a_nudge = hex_frac_cell{a.q + 0.000001f,
353 a.r + 0.000001f,
354 a.s - 0.000002f};
355 auto const b_nudge = hex_frac_cell{b.q + 0.000001f,
356 b.r + 0.000001f,
357 b.s - 0.000002f};
358
359 ui32 const n = hex_distance(a, b);
360 auto const step = 1.f / std::max(n, 1u);
361
362 hex_cell::list results;
363 for (auto i = 0u; i <= n; ++i)
364 results.push_back(hex_round(hex_lerp(a_nudge,
365 b_nudge,
366 step * i)));
367
368 return results;
369}

513inline hex_point hex_to_pixel(hex_layout const& layout,
514 hex_cell const& cell) {
515 auto const& m = layout.orientation;
516 auto const& size = layout.size;
517 auto const& origin = layout.origin;
518 auto const x = (m.f0 * cell.q + m.f1 * cell.r) * size.x;
519 auto const y = (m.f2 * cell.q + m.f3 * cell.r) * size.y;

529inline hex_frac_cell hex_pixel_to_cell(hex_layout const& layout,
530 hex_point const& p) {
531 auto const& m = layout.orientation;
532 auto const& size = layout.size;
533 auto const& origin = layout.origin;
534 auto const pt = hex_point{(p.x - origin.x) / size.x,
535 (p.y - origin.y) / size.y};
536 auto const q = m.b0 * pt.x + m.b1 * pt.y;
537 auto const r = m.b2 * pt.x + m.b3 * pt.y;
538 return {q, r, -q - r};
